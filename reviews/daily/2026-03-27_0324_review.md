### 改进 3: K 线分析强制确认

**新增流程**:
```python
# 开仓前必跑 K 线分析
def pre_trade_check(symbol):
    # 运行 K 线分析
    kline_result = run_kline_analysis(symbol)
    
    # 获取交易信号
    trade_signal = get_trade_signal(symbol)
    
    # 对比两者
    if kline_result['signal'] != trade_signal['direction']:
        return False, "K 线分析与信号冲突，观望"
    
    # 检查 K 线形态
    if kline_result['pattern'] in ['doji', 'spinning_top']:
        return False, "K 线形态不明，观望"
    
    # 检查支撑/阻力
    distance_to_support = kline_result['distance_to_support']
    distance_to_resistance = kline_result['distance_to_resistance']
    
    if trade_signal['direction'] == 'BUY' and distance_to_resistance < 10:
        return False, "接近阻力位，风险高"
    
    if trade_signal['direction'] == 'SELL' and distance_to_support < 10:
        return False, "接近支撑位，风险高"
    
    return True, "K 线确认通过"
```
